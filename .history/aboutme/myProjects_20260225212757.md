This is a fully programmed collatz conjecture generator in C#.
```cs
static void collatzconjecture ()
{
    Int32 iterations = 0, total = 0, initialvalue, over100 = 0, between1and100 = 0;
    Boolean goodval = false;
    do
    {
        Console.Clear();
        Console.WriteLine("COLLATZ CONJECTURE");
        Console.WriteLine("Please input an integer.");
        goodval = Int32.TryParse(Console.ReadLine(), out initialvalue);

        if (goodval == false)
        {
            Console.WriteLine("That's not an integer. Press any key to restart.");
            Console.ReadKey();

        }
    } while (goodval == false);
    total = initialvalue;

    while (total != 1) // while Collatz Conjecture is running, everything below happens
    {
        if (total % 2 == 0) // this is asking: is the number EVEN?
        {
            total = total / 2; // this happens when the number is EVEN
        }
        else // this asks is the number ODD?
        {
            total = total * 3 + 1; // this happens when the number is ODD
        }
        
        if (total > 100)
        {
            over100++;
        }
        else
        {
            between1and100++;
        }
        iterations++;
        Console.WriteLine(total);
    }

    Console.WriteLine("You inputted " + initialvalue + ".");
    Console.WriteLine("The program ran for " + iterations + " iterations until 1 was reached.");
    Console.WriteLine("During the loop, the number exceeded 100 " + over100 + " times.");
    Console.WriteLine("And the number was between 1 and 100 " + between1and100 + " times.");
    Console.WriteLine("Press any key to return to the menu...");
    Console.ReadKey();
}
```

This is a simple game where you have to add coins to make $1.
```cs
static void dollargame ()
{
    Boolean success = false, goodval1, goodval2, goodval3, goodval4;
    Int32 pennies = 0, nickels = 0, dimes, quarters, totalturns = 0;
    Single money = 0.00f;
    Single rounded = 0.00f;
    
    while (success == false) // until the user makes exactly one dollar, everything below happens
    {
        Console.Clear();
        Console.WriteLine("DOLLAR GAME");
        Console.WriteLine("Please input the number of each coin type you have. The goal is to make exactly one dollar.");

        money = 0.00f;
        Console.WriteLine("How many pennies do you have?");
        goodval1 = Int32.TryParse(Console.ReadLine(), out pennies);

        Console.WriteLine("How many nickels do you have?");
        goodval2 = Int32.TryParse(Console.ReadLine(), out nickels);

        Console.WriteLine("How many dimes do you have?");
        goodval3 = Int32.TryParse(Console.ReadLine(), out dimes);

        Console.WriteLine("How many quarters do you have?");
        goodval4 = Int32.TryParse(Console.ReadLine(), out quarters);

        if (goodval1 == false || goodval2 == false || goodval3 == false || goodval4 == false)
        {
            Console.WriteLine("One or more of the inputs was not a integer. Try again. Press any key to return to the beginning...");
            Console.ReadKey();
            continue;
        }

        money = (pennies * 0.01f) + (nickels * 0.05f) + (dimes * 0.10f) + (quarters * 0.25f);

        rounded = MathF.Round(money, 2);

        totalturns++;

        if (rounded > 1.00f)
        {
            success = false;
            rounded = rounded - 1.00f;
            rounded = MathF.Round(rounded, 2);
            Console.WriteLine("Sorry, you were $" + rounded + " over $1. Try again. Press any key to continue...");
            Console.ReadKey();
            continue;
        }

        if (rounded < 1.00f)
        {
            success = false;
            rounded = 1.00f - rounded;
            rounded = MathF.Round(rounded, 2);
            Console.WriteLine("Sorry, you were $" + rounded + " below $1. Try again. Press any key to continue...");
            Console.ReadKey();
            continue;
        }

        if (rounded == 1.00f)
        { 
            success = true;
        }
    }
    Console.WriteLine("Congratulations! You made $1. It took you " + totalturns + " turns.");
    Console.WriteLine("Press any key to return to the menu...");
    Console.ReadKey();
}
```
